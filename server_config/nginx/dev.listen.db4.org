server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name dev.listen.dbp4.org www.dev.listen.dbp4.org;
    root /home/ubuntu/dbp_4_reader/build;

    # FORGE SSL (DO NOT REMOVE!)
    ssl_certificate /etc/nginx/ssl/dev-listen.dbp4.org/server.crt;
    ssl_certificate_key /etc/nginx/ssl/dev-listen.dbp4.org/server.key;

    ssl_protocols TLSv1.1 TLSv1.2;  #kurt removed TLSv1
    ssl_prefer_server_ciphers on;
    ssl_dhparam /etc/nginx/dhparams.pem;

    add_header X-Frame-Options "SAMEORIGIN";
    add_header X-XSS-Protection "1; mode=block";
    add_header X-Content-Type-Options "nosniff";

    add_header Strict-Transport-Security "max-age=31536000; includeSubdomains; preload";

    index index.html;

    charset utf-8;

    location = /favicon.ico { access_log off; log_not_found off; }
    location = /robots.txt  { access_log off; log_not_found off; }

    location /sw.js {
        add_header Cache-Control "no-cache";

        proxy_cache_bypass $http_pragma;
        proxy_cache_revalidate on;
        expires off;
        access_log off;
    }

    access_log off;
    error_log  /var/log/nginx/dbp_4_reader-error.log error;

    error_page 404 /pages/_error.js;

    location / {
        proxy_pass http://localhost:3000;
    }

    location ~ /\.(?!well-known).* {
        deny all;
    }
}
